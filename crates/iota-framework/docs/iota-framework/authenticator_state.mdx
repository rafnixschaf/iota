</code></pre>



<details>
<summary>Implementation</summary>


<pre><code>
<b>fun</b> <Link to="../iota-framework/authenticator_state#0x2_authenticator_state_string_bytes_lt">string_bytes_lt</Link>(a: &String, b: &String): bool \{
    <b>let</b> a_bytes = a.bytes();
    <b>let</b> b_bytes = b.bytes();

    <b>if</b> (a_bytes.length() &lt; b_bytes.length()) \{
        <b>true</b>
    } <b>else</b> <b>if</b> (a_bytes.length() &gt; b_bytes.length()) \{
        <b>false</b>
    } <b>else</b> \{
        <b>let</b> <b>mut</b> i = 0;
        <b>while</b> (i &lt; a_bytes.length()) \{
            <b>let</b> a_byte = a_bytes[i];
            <b>let</b> b_byte = b_bytes[i];
            <b>if</b> (a_byte &lt; b_byte) \{
                <b>return</b> <b>true</b>
            } <b>else</b> <b>if</b> (a_byte &gt; b_byte) \{
                <b>return</b> <b>false</b>
            };
            i = i + 1;
        };
        // all bytes are equal
        <b>false</b>
    }
}
</code></pre>

    new_active_jwks: <Link to="../move-stdlib/vector#0x1_vector">vector</Link>&lt;<Link to="../iota-framework/authenticator_state#0x2_authenticator_state_ActiveJwk">ActiveJwk</Link>&gt;,

    <b>assert</b>!(ctx.sender() == @0x0, <Link to="../iota-framework/authenticator_state#0x2_authenticator_state_ENotSystemAddress">ENotSystemAddress</Link>);

    <Link to="../iota-framework/authenticator_state#0x2_authenticator_state_check_sorted">check_sorted</Link>(&new_active_jwks);
    <b>let</b> new_active_jwks = <Link to="../iota-framework/authenticator_state#0x2_authenticator_state_deduplicate">deduplicate</Link>(new_active_jwks);

    <b>let</b> inner = self.<Link to="../iota-framework/authenticator_state#0x2_authenticator_state_load_inner_mut">load_inner_mut</Link>();

    <b>let</b> <b>mut</b> res = <Link to="../move-stdlib/vector#0x1_vector">vector</Link>[];
    <b>let</b> <b>mut</b> i = 0;
    <b>let</b> <b>mut</b> j = 0;
    <b>let</b> active_jwks_len = inner.active_jwks.length();
    <b>let</b> new_active_jwks_len = new_active_jwks.length();

    <b>while</b> (i &lt; active_jwks_len && j &lt; new_active_jwks_len) \{
        <b>let</b> old_jwk = &inner.active_jwks[i];
        <b>let</b> new_jwk = &new_active_jwks[j];

        // when they are equal, push only one, but <b>use</b> the max epoch of the two
        <b>if</b> (<Link to="../iota-framework/authenticator_state#0x2_authenticator_state_active_jwk_equal">active_jwk_equal</Link>(old_jwk, new_jwk)) \{
            <b>let</b> <b>mut</b> jwk = *old_jwk;
            jwk.epoch = <Link to="../iota-framework/math#0x2_math_max">math::max</Link>(old_jwk.epoch, new_jwk.epoch);
            res.push_back(jwk);
            i = i + 1;
            j = j + 1;
        } <b>else</b> <b>if</b> (<Link to="../iota-framework/authenticator_state#0x2_authenticator_state_jwk_id_equal">jwk_id_equal</Link>(&old_jwk.jwk_id, &new_jwk.jwk_id)) \{
            // <b>if</b> only jwk_id is equal, then the key <b>has</b> changed. Providers should not send
            // JWKs like this, but <b>if</b> they do, we must ignore the new <Link to="../iota-framework/authenticator_state#0x2_authenticator_state_JWK">JWK</Link> <b>to</b> avoid having a
            // liveness / forking issues
            res.push_back(*old_jwk);
            i = i + 1;
            j = j + 1;
        } <b>else</b> <b>if</b> (<Link to="../iota-framework/authenticator_state#0x2_authenticator_state_jwk_lt">jwk_lt</Link>(old_jwk, new_jwk)) \{
            res.push_back(*old_jwk);
            i = i + 1;
        } <b>else</b> \{
            res.push_back(*new_jwk);
            j = j + 1;
        }
    };

    <b>while</b> (i &lt; active_jwks_len) \{
        res.push_back(inner.active_jwks[i]);
        i = i + 1;
    };
    <b>while</b> (j &lt; new_active_jwks_len) \{
        res.push_back(new_active_jwks[j]);
        j = j + 1;
    };

    inner.active_jwks = res;
}

    // and store the counts in a <Link to="../move-stdlib/vector#0x1_vector">vector</Link> that parallels the (sorted) active_jwks <Link to="../move-stdlib/vector#0x1_vector">vector</Link>
    <b>let</b> <b>mut</b> issuer_max_epochs = <Link to="../move-stdlib/vector#0x1_vector">vector</Link>[];
    <b>let</b> <b>mut</b> i = 0;

    <b>while</b> (i &lt; len) \{
        <b>let</b> cur = &inner.active_jwks[i];
        <b>let</b> cur_iss = &cur.jwk_id.iss;
        <b>if</b> (prev_issuer.is_none()) \{
            prev_issuer.fill(*cur_iss);
            issuer_max_epochs.push_back(cur.epoch);

            <b>if</b> (cur_iss == prev_issuer.borrow()) \{
                <b>let</b> back = issuer_max_epochs.length() - 1;
                <b>let</b> prev_max_epoch = &<b>mut</b> issuer_max_epochs[back];
                *prev_max_epoch = <Link to="../iota-framework/math#0x2_math_max">math::max</Link>(*prev_max_epoch, cur.epoch);
            } <b>else</b> \{
                *prev_issuer.borrow_mut() = *cur_iss;
                issuer_max_epochs.push_back(cur.epoch);
            }
        };
        i = i + 1;
    };

    // Now, filter out any JWKs that are below the min_epoch, unless that issuer <b>has</b> no
    // JWKs &gt;= the min_epoch, in which case we keep all of them.

    <b>let</b> <b>mut</b> i = 0;
    <b>let</b> <b>mut</b> j = 0;

            *prev_issuer.borrow_mut() = *cur_iss;
            j = j + 1;
        };

        <b>let</b> max_epoch_for_iss = &issuer_max_epochs[j];

        i = i + 1;
    };
