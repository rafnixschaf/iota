<b>public</b>(package) <b>fun</b> <Link to="stake_subsidy#0x3_stake_subsidy_advance_epoch">advance_epoch</Link>(self: &<b>mut</b> <Link to="stake_subsidy#0x3_stake_subsidy_StakeSubsidy">StakeSubsidy</Link>): Balance&lt;IOTA&gt; \{
    // Take the minimum of the reward amount and the remaining <Link to="../iota-framework/balance#0x2_balance">balance</Link> in
    // order <b>to</b> ensure we don't overdraft the remaining stake subsidy
    // <Link to="../iota-framework/balance#0x2_balance">balance</Link>
    <b>let</b> to_withdraw = <Link to="../iota-framework/math#0x2_math_min">math::min</Link>(self.current_distribution_amount, self.<Link to="../iota-framework/balance#0x2_balance">balance</Link>.value());

    // Drawn down the subsidy for this epoch.
    <b>let</b> <Link to="stake_subsidy#0x3_stake_subsidy">stake_subsidy</Link> = self.<Link to="../iota-framework/balance#0x2_balance">balance</Link>.split(to_withdraw);

    self.distribution_counter = self.distribution_counter + 1;

    // Decrease the subsidy amount only when the current period ends.
    <b>if</b> (self.distribution_counter % self.stake_subsidy_period_length == 0) \{
        <b>let</b> decrease_amount = self.current_distribution_amount <b>as</b> u128
            * (self.stake_subsidy_decrease_rate <b>as</b> u128) / <Link to="stake_subsidy#0x3_stake_subsidy_BASIS_POINT_DENOMINATOR">BASIS_POINT_DENOMINATOR</Link>;
        self.current_distribution_amount = self.current_distribution_amount - (decrease_amount <b>as</b> u64)
    };

    <Link to="stake_subsidy#0x3_stake_subsidy">stake_subsidy</Link>
}
